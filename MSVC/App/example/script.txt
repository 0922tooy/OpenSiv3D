
void Main()
{
 	Image image(480, 480, Palette::White);
 
 	image.save("s.png");

 	DynamicTexture texture(image);
 
 	while (System::Update())
 	{
 		if (MouseL.pressed())
 		{
 			Line(Cursor::PreviousPos(), Cursor::Pos()).overwrite(image, 8, Palette::Orange);
 
 			texture.fill(image);
 		}
 
 		texture.draw();
 	}
}
